


cd /home/mike0/Desktop
tar zxvf demo.tar.gz    #tar zcvf tmp.tar.gz tmp   #打包并且以gzip格式压缩


cd ~/Desktop/demo/fabric-samples/first-network/
//关闭项目并删除volume卷
./byfn.sh -m down -v

//启动流程：
//生成证书
./byfn.sh -m generate
//启动first-network项目
./byfn.sh -m up
//安装walletcc项目
sh installWalletcc.sh 1.0
//启动CA
docker-compose -f docker-compose-e2e.yaml up -d
//进入cli容器
docker exec -it cli bash
//创建钱包
// args: 0-ID
peer chaincode invoke -n walletcc -c '{"Args":["createWallet","1"]}' -C mychannel --tls true --cafile /opt/gopath/src/github.com/hyperledger/fabric/peer/crypto/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

peer chaincode invoke -n walletcc -c '{"Args":["createWallet","2"]}' -C mychannel --tls true --cafile /opt/gopath/src/github.com/hyperledger/fabric/peer/crypto/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

//交易
//args:0-fromWalletID,1-toWalletID,2-PropertyName,3-Digital，4-Cash，5-Times
//钱包1 转给 钱包2 RMB, digital, 100， 20190102-132221
peer chaincode invoke -n walletcc -c '{"Args":["transfer","1","2","RMB","100","0","20190102-132221"]}' -C mychannel --tls true --cafile /opt/gopath/src/github.com/hyperledger/fabric/peer/crypto/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

//钱包2 转给 钱包1 JPY, digital, -100， 20190102-154359
peer chaincode invoke -n walletcc -c '{"Args":["transfer","2","1","GBP","-100","0","20190102-154359"]}' -C mychannel --tls true --cafile /opt/gopath/src/github.com/hyperledger/fabric/peer/crypto/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

//查询钱包
//args: 0-ID
peer chaincode query -n walletcc -c '{"Args":["queryWallet","1"]}' -C mychannel

peer chaincode query -n walletcc -c '{"Args":["queryWallet","2"]}' -C mychannel

//查询时间点钱包状态
// args: 0-ID,1-times
peer chaincode query -n walletcc -c '{"Args":["queryWalletByTime","1","20190102-132221"]}' -C mychannel

peer chaincode query -n walletcc -c '{"Args":["queryWalletByTime","1","20190102-154359"]}' -C mychannel

//查询钱包资产在某个时间段的交易记录
// args: 0-ID，1-time1,2-time2,3-PropertyName
peer chaincode query -n walletcc -c '{"Args":["queryHistoryByTimeQuantum","1","20190102-132221","20190102-154359","RMB"]}' -C mychannel


// 启动node-sdk项目
cd ~/Desktop/demo/node-sdk
npm install
// npm rebuild
cp ~/Desktop/demo/fabric-samples/first-network/crypto-config ./ -r

node query.js
//查询某一个时间点钱包状态
node queryWalletByTime.js
//查询钱包资产在某个时间段的交易记录
node queryHistoryByTimeQuantum.js
